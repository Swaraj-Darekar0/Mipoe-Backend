create table public.creator (
  id uuid references auth.users not null primary key,
  username character varying(80) not null,
  email character varying(120) not null,
  password_hash character varying(128) not null,
  nickname character varying(80) null,
  bio text null,
  phone character varying(20) null,
  join_date date not null,
  profile_completed boolean null default false,
  wallet_balance double precision null default 0.0,
  payout_method character varying(10) null default null::character varying,
  upi_id character varying(255) null default null::character varying,
  bank_account character varying(20) null default null::character varying,
  ifsc character varying(11) null default null::character varying,
  account_holder_name character varying(255) null default null::character varying,
  notifications jsonb[] null default '{}'::jsonb[],
  constraint creator_pkey primary key (id),
  constraint creator_email_key unique (email),
  constraint creator_username_key unique (username)
) TABLESPACE pg_default;
create index IF not exists idx_creator_wallet on public.creator using btree (id, wallet_balance) TABLESPACE pg_default;
create index IF not exists idx_creator_payout_method on public.creator using btree (payout_method) TABLESPACE pg_default;

create table public.campaign (
  id serial not null,
  brand_id integer not null,
  platform character varying(50) not null,
  budget double precision not null,
  cpv double precision not null,
  hashtag character varying(100) null,
  audio character varying(200) null,
  deadline date not null,
  name character varying(120) not null,
  is_active boolean null default true,
  total_view_count integer null default 0,
  requirements text null,
  view_threshold integer null default 0,
  asset_link character varying null,
  category character varying(20) not null default 'fashion_clothing'::character varying,
  image_url text null,
  funds_allocated double precision null default 0.0,
  funds_distributed double precision null default 0.0,
  constraint campaign_pkey primary key (id),
  constraint fk_brand foreign KEY (brand_id) references brand (id) on delete CASCADE,
  constraint check_funds_distributed check ((funds_distributed <= funds_allocated)),
  constraint check_image_url_format check ((image_url ~* '^https?://.*'::text)),
  constraint valid_category check (
    (
      (category)::text = any (
        array[
          ('fashion_clothing'::character varying)::text,
          ('beauty_products'::character varying)::text,
          ('youtube'::character varying)::text
        ]
      )
    )
  )
) TABLESPACE pg_default;
create index IF not exists idx_campaign_funds_allocated on public.campaign using btree (id, funds_allocated) TABLESPACE pg_default;
create index IF not exists idx_campaign_funds_distributed on public.campaign using btree (id, funds_distributed) TABLESPACE pg_default;

//Brand Transaction view
 SELECT bt.brand_id,
    NULL::integer AS creator_id,
    bt.campaign_id,
    bt.created_at,
    bt.description,
    bt.type::text AS type,
    bt.amount,
    bt.status,
    bt.id,
    NULL::text AS payout_method,
    NULL::text AS utr,
    bt.external_txn_id,
    NULL::text AS creator_username,
    NULL::text AS campaign_name
   FROM brand_transactions bt
UNION ALL
 SELECT c.brand_id,
    ct.creator_id,
    ct.campaign_id,
    ct.created_at,
    ct.description,
    'distribution'::text AS type,
    - ct.amount AS amount,
    ct.status,
    ct.id,
    ct.payout_method,
    ct.utr,
    ct.external_txn_id,
    NULL::text AS creator_username,
    NULL::text AS campaign_name
   FROM creator_transactions ct
     JOIN campaign c ON ct.campaign_id = c.id
  WHERE ct.type = 'earning'::creator_transaction_type;


  //detailed Brand creator Trnsaction view 
   SELECT b_view.brand_id,
    b_view.creator_id,
    b_view.campaign_id,
    b_view.created_at,
    b_view.description,
    b_view.type,
    b_view.amount,
    b_view.status,
    b_view.id,
    b_view.external_txn_id,
    COALESCE(c.name, 'N/A'::character varying) AS campaign_name,
    COALESCE(cr.username, 'N/A'::character varying) AS creator_username
   FROM brand_transaction_view b_view
     LEFT JOIN campaign c ON b_view.campaign_id = c.id
     LEFT JOIN creator cr ON b_view.creator_id = cr.id;



///refund summary view 
 SELECT c.id AS campaign_id,
    c.name AS campaign_name,
    b.id AS brand_id,
    count(ra.id) FILTER (WHERE ra.status::text = 'completed'::text) AS total_refunds_completed,
    COALESCE(sum(ra.approved_amount) FILTER (WHERE ra.status::text = 'completed'::text), 0::numeric) AS total_refunded_amount,
    count(ra.id) FILTER (WHERE ra.status::text = 'pending'::text) AS pending_refund_requests,
    COALESCE(sum(ra.requested_amount) FILTER (WHERE ra.status::text = 'pending'::text), 0::numeric) AS pending_refund_total
   FROM campaign c
     JOIN brand b ON c.brand_id = b.id
     LEFT JOIN refund_audits ra ON c.id = ra.campaign_id
  GROUP BY c.id, c.name, b.id;
